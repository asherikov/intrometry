54 template <class t_Message>
55 class Subscription
56 {
57 public:
58 typename rclcpp::Subscription<t_Message>::SharedPtr subscription_;
59 std::atomic<std::size_t> counter_ = 0;
60
61 public:
62 void subscribe(std::shared_ptr<rclcpp::Node> &node, const std::string &topic)
63 {
64 subscription_ = node->create_subscription<t_Message>(
65 topic,
66 rclcpp::QoS(/*history_depth=*/10).best_effort(),
67 [this](const t_Message & /*msg*/) { ++counter_; });
68 }
69
70 [[nodiscard]] bool atLeastOne() const
71 {
72 return (counter_ > 0);
73 }
74 };

94 {

96
97 names_subscription_.subscribe(node_, std::string("/intrometry/") + topic + "/names");
98 values_subscription_.subscribe(node_, std::string("/intrometry/") + topic + "/values");
99
100 this->addSupervisedThread(
101 tut::thread::Parameters(
102 tut::thread::Parameters::Restart(/*attempts=*/1, /*sleep_ms=*/0),
103 tut::thread::Parameters::TerminationPolicy::KILLALL,
104 tut::thread::Parameters::ExceptionPolicy::PASS),
105 &SubscriberNode::spin,
106 this);
107 }
108
109
110 [[nodiscard]] bool checkReceived() const
111 {
112 return (names_subscription_.atLeastOne() and values_subscription_.atLeastOne());
113 }

122 void spin()
123 {
124 rclcpp::WallRate loop_rate(1000);
125
126 while (rclcpp::ok() and not isThreadSupervisorInterrupted())
127 {
128 rclcpp::spin_some(node_);
